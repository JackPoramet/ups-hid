echo 'SUBSYSTEM=="usb", ATTRS{idVendor}=="06da", ATTRS{idProduct}=="ffff", MODE="0666"' | sudo tee /etc/udev/rules.d/99-ups.rules
echo 'KERNEL=="hidraw*", SUBSYSTEM=="hidraw", ATTRS{idVendor}=="06da", ATTRS{idProduct}=="ffff", MODE="0666"' | sudo tee -a /etc/udev/rules.d/99-ups.rules
